1   package avaruustaistelu.objektit;
2   
3   import avaruustaistelu.tiedostojenkasittelija.TiedostojenKasittelija;
4   import java.awt.Graphics;
5   import java.awt.Rectangle;
6   import java.awt.image.BufferedImage;
7   
8   /**
9    * Luokka, joka sisältää elämäpakettiin liittyvät tiedot ja metodit.
10   * 
11   * @author Jyri
12   */
13  public class Elamapaketti extends Objekti {
14  
15      private BufferedImage elamapaketinKuva;
16      
17      /**
18       * Luokan konstruktori, joka saa parametrikseen elämäpaketin x- ja y-koordinaatit.
19       * 
20       * @param x x-koordinaatti
21       * @param y y-koordinaatti
22       */
23      public Elamapaketti(int x, int y) {
24          super(x, y);
25      }
26      
27      /**
28       * Liikuttaa elämäpakettia.
29       */
30      public void liiku() {
31          this.y = this.y + 3;
32      }
33      
34      /**
35       * Piirtää elämäpaketin kuvan perusteella, joka asetetaan setElamapaketinKuva -metodin avulla Avaruustaistelu luokassa 
36       * Tiedostojenkäsittelijältä.
37       * 
38       * @param g Elämäpaketin kuva
39       */
40      @Override
41      public void piirra(Graphics g) {
42          g.drawImage(this.elamapaketinKuva, this.x, this.y, null);
43      }
44  
45      /**
46       * Palauttaa sen alueen, missä elämäpaketti sillä hetkellä sijaitsee. Tätä tietoa käytetään,
47       * kun tutkitaan koskettavatko eri objektit toisiaan "intersects" -metodin avulla.
48       * 
49       * @return Rectangle Elämäpaketin sijainnin alue
50       */
51      @Override
52      public Rectangle getObjektinSijainninAlue() {
53          return new Rectangle(this.x, this.y, 20, 20);
54      }
55      
56      public void setElamapaketinKuva(BufferedImage elamapaketinKuva) {
57          this.elamapaketinKuva = elamapaketinKuva;
58      }
59      
60  }
