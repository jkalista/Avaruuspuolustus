1   package avaruustaistelu.objektit;
2   
3   import java.awt.Color;
4   import java.awt.Graphics;
5   import java.awt.Rectangle;
6   
7   /**
8    * Luokka sisältää pelaajan ohjaaman avaruusaluksen ampumaan ohjukseen liittyvät tiedot ja metodit.
9    * 
10   * @author Jyri
11   */
12  public class Ohjus extends Objekti {
13  
14      public Ohjus(int x, int y) {
15          super(x, y);
16      }
17      
18      /**
19      * Liikuttaa ohjusta
20      */
21      public void liiku() {
22          this.y = this.y - 5;
23      }
24  
25      /**
26      * Piirtää ohjuksen, eli siihen kuuluvat harmaan ja punaisen alueen.
27      * 
28      * @param g Ohjuksen grafiikka
29      */
30      @Override
31      public void piirra(Graphics g) {
32          g.setColor(Color.GRAY);
33          g.fillRect(x, y, 5, 10);
34          g.setColor(Color.RED);
35          g.fillRect(x, y + 10, 5, 5);
36      }
37  
38      /**
39      * Palauttaa sen alueen, missä ohjus sillä hetkellä sijaitsee. Tätä tietoa käytetään,
40      * kun tutkitaan koskettavatko eri objektit toisiaan "intersects" -metodin avulla.
41      * 
42      * @return Rectangle Ohjuksen sijainnin alue
43      */
44      @Override
45      public Rectangle getObjektinSijainninAlue() {
46          return new Rectangle(this.x, this.y, 5, 15);
47      }
48      
49  }
