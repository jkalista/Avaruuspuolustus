269     private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
270         // TODO add your handling code here:
271         
272         SwingWorker<Void, Void> swingWorker = new SwingWorker<Void, Void>() {
273             
274             public void sleep(int aikaMillisekunteina) {
275                 try {
276                     Thread.sleep(aikaMillisekunteina);
277                 } catch (InterruptedException ex) {
278                     Logger.getLogger(MenuValikko.class.getName()).log(Level.SEVERE, null, ex);
279                 }
280             }
281             
282             @Override
283             protected Void doInBackground() throws Exception {
284                 Avaruustaistelu avaruustaistelu = new Avaruustaistelu();
285                 Kayttoliittyma kayttoliittyma = new Kayttoliittyma(avaruustaistelu);
286                 SwingUtilities.invokeLater(kayttoliittyma);
287                 
288                 while (kayttoliittyma.getPiirtoalusta() == null) {
289                     sleep(100);
290                 }
291                 
292                 avaruustaistelu.setPiirtoalusta(kayttoliittyma.getPiirtoalusta());
293                 avaruustaistelu.pelinLoop();
294                 return null;
295             }
296             
297         };
298         
299         swingWorker.execute();
300     }//GEN-LAST:event_jButton1ActionPerformed

357     public void paivitaEnnatykset() {
358         File tiedosto = new File("ennatykset.txt");
359         
360         try {
361             Scanner lukija = new Scanner(tiedosto.getAbsoluteFile());
362             int ensimmainenSija = Integer.parseInt(lukija.nextLine());
363             int toinenSija = Integer.parseInt(lukija.nextLine());
364             int kolmasSija = Integer.parseInt(lukija.nextLine());
365             this.jLabel2Eka.setText("1. sija: " + ensimmainenSija);
366             this.jLabel3Toka.setText("2. sija: " + toinenSija);
367             this.jLabel4Kolmas.setText("3. sija: " + kolmasSija);
368         } catch (FileNotFoundException | NumberFormatException e) {
369             System.out.print(e);
370         }
371     }
