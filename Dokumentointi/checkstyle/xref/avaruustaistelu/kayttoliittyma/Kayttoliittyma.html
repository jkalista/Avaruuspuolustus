1   package avaruustaistelu.kayttoliittyma;
2   
3   import avaruustaistelu.avaruustaistelu.Avaruustaistelu;
4   import java.awt.Container;
5   import java.awt.Dimension;
6   import javax.swing.JFrame;
7   import javax.swing.WindowConstants;
8   
9   /**
10   * Luokka, jossa luodaan käyttöliittymä.
11   * 
12   * @author Jyri
13   */
14  public class Kayttoliittyma implements Runnable {
15  
16      private JFrame pelinKehykset;
17      private final Avaruustaistelu avaruustaistelu;
18      private Piirtoalusta piirtoalusta;
19      
20      public Kayttoliittyma(Avaruustaistelu avaruustaistelu) {
21          this.avaruustaistelu = avaruustaistelu;
22      }
23      
24      @Override
25      public void run() {
26          this.pelinKehykset = new JFrame("Avaruuspuolustus");
27          this.pelinKehykset.setPreferredSize(new Dimension(650, 900));
28  
29          this.pelinKehykset.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
30  
31          luoKomponentit(this.pelinKehykset.getContentPane());
32  
33          this.pelinKehykset.pack();
34          this.pelinKehykset.setResizable(false);
35          this.pelinKehykset.setVisible(true);
36      }
37      
38      /**
39      * Luo piirtoalustan Avaruuspuolustus pelille ja lisää näppäimistönkuuntelijan
40      * Avaruuspuolustus pelin tuntemalle pelaajalle.
41      * 
42      * @param container Container
43      */
44      private void luoKomponentit(Container container) {
45          this.piirtoalusta = new Piirtoalusta(this.avaruustaistelu);
46          container.add(this.piirtoalusta);
47           
48          this.pelinKehykset.addKeyListener(new NappaimistonKuuntelija(this.avaruustaistelu.getAvaruusalus()));
49      }
50  
51      public JFrame getFrame() {
52          return this.pelinKehykset;
53      }
54      
55      public Piirtoalusta getPiirtoalusta() {
56          return this.piirtoalusta;
57      }
58      
59  }
