1   package avaruustaistelu.kayttoliittyma;
2   
3   import avaruustaistelu.avaruustaistelu.Avaruustaistelu;
4   import java.awt.Container;
5   import java.awt.Dimension;
6   import javax.swing.JFrame;
7   import javax.swing.WindowConstants;
8   
9   /**
10   * Luokka, jossa luodaan pelin käyttöliittymä.
11   * 
12   * @author Jyri
13   */
14  public class Kayttoliittyma implements Runnable {
15  
16      private JFrame pelinKehykset;
17      private final Avaruustaistelu avaruustaistelu;
18      private Piirtoalusta piirtoalusta;
19      
20      /**
21       * Luokan konstruktori, joka saa parametrikseen Avaruustaistelu luokan ilmentymän.
22       * 
23       * @param avaruustaistelu Avaruustaistelu peli
24       */
25      public Kayttoliittyma(Avaruustaistelu avaruustaistelu) {
26          this.avaruustaistelu = avaruustaistelu;
27      }
28      
29      @Override
30      public void run() {
31          this.pelinKehykset = new JFrame("Avaruustaistelu");
32          this.pelinKehykset.setPreferredSize(new Dimension(650, 900));
33  
34          this.pelinKehykset.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
35  
36          luoKomponentit(this.pelinKehykset.getContentPane());
37  
38          this.pelinKehykset.pack();
39          this.pelinKehykset.setResizable(false);
40          this.pelinKehykset.setVisible(true);
41      }
42      
43      /**
44       * Luo piirtoalustan Avaruustaistelu pelille ja lisää näppäimistönkuuntelijan
45       * Avaruustaistelu pelin tuntemalle pelaajalle.
46       * 
47       * @param container Container
48       */
49      private void luoKomponentit(Container container) {
50          this.piirtoalusta = new Piirtoalusta(this.avaruustaistelu);
51          container.add(this.piirtoalusta);
52          
53          this.pelinKehykset.addKeyListener(new NappaimistonKuuntelija(this.avaruustaistelu, this.pelinKehykset));
54      }
55  
56      public JFrame getFrame() {
57          return this.pelinKehykset;
58      }
59      
60      public Piirtoalusta getPiirtoalusta() {
61          return this.piirtoalusta;
62      }
63      
64  }
