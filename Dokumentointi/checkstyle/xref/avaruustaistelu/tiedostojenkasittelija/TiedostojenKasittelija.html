1   package avaruustaistelu.tiedostojenkasittelija;

3   import java.awt.image.BufferedImage;
4   import java.io.File;
5   import java.io.FileWriter;
6   import java.io.IOException;
7   import java.util.Scanner;
8   import javax.imageio.ImageIO;
9   
10  /**
11   * Luokka, jossa tapahtuu kaikki tiedostojen käsittelyyn liittyvät asiat, esimerkiksi
12   * kuvien hakeminen.
13   * 
14   * @author Jyri
15   */
16  public class TiedostojenKasittelija {
17      
18      private final BufferedImage avaruusaluksenKuva;
19      private final BufferedImage elamapaketinKuva;
20      
21      /**
22       * Luokan konstruktori, jossa haetaan tiedostoista peliin liittyvät kuvat.
23       */
24      public TiedostojenKasittelija() {
25          BufferedImage avaruusalusKuva = null;
26          try {
27              avaruusalusKuva = ImageIO.read(this.getClass().getResource("/tiedostot/Alus.png"));
28          } catch (IOException e) {
29              System.out.print(e);
30          }
31          this.avaruusaluksenKuva = avaruusalusKuva;
32          
33          BufferedImage elamapakettiKuva = null;
34          try {
35              elamapakettiKuva = ImageIO.read(this.getClass().getResource("/tiedostot/Elamapaketti.png"));
36          } catch (IOException e) {
37              System.out.print(e);
38          }
39          this.elamapaketinKuva = elamapakettiKuva;
40      }
41      
42      /**
43       * Palauttaa pelaajan ohjaaman avaruusaluksen kuvan.
44       * 
45       * @return BufferedImage Avaruusaluksen kuva
46       */
47      public BufferedImage getAvaruusaluksenKuva() {
48          return this.avaruusaluksenKuva;
49      }
50      
51      /**
52       * Palauttaa elämäpaketin kuvan.
53       * 
54       * @return BufferedImage Elamapaketin kuva
55       */
56      public BufferedImage getElamapaketinKuva() {
57          return this.elamapaketinKuva;
58      }
59      
60      /**
61       * Metodi, joka tarkistaa riittääkö pelistä saadut pisteet ennätyspisteluetteloon.
62       * 
63       * @param saadutPisteet Pisteet pelistä
64       */
65      public void tarkastaRiittavatkoPisteetEnnatyslistalle(int saadutPisteet) {
66          File tiedosto = new File("ennatykset.txt");
67          
68          try {
69              Scanner lukija = new Scanner(tiedosto);
70              
71              int ensimmainenSija = Integer.parseInt(lukija.nextLine());
72              int toinenSija = Integer.parseInt(lukija.nextLine());
73              int kolmasSija = Integer.parseInt(lukija.nextLine());
74              
75              FileWriter kirjoittaja = new FileWriter(tiedosto);
76              
77              if (saadutPisteet > ensimmainenSija) {
78                  kirjoittaja.write(Integer.toString(saadutPisteet) + System.lineSeparator());
79                  kirjoittaja.write(Integer.toString(ensimmainenSija) + System.lineSeparator());
80                  kirjoittaja.write(Integer.toString(toinenSija));
81              } else {
82                  kirjoittaja.write(Integer.toString(ensimmainenSija) + System.lineSeparator());
83                  
84                  if (saadutPisteet > toinenSija) {
85                      kirjoittaja.write(Integer.toString(saadutPisteet) + System.lineSeparator());
86                      kirjoittaja.write(Integer.toString(toinenSija));
87                  } else {
88                      kirjoittaja.write(Integer.toString(toinenSija) + System.lineSeparator());
89                      
90                      if (saadutPisteet > kolmasSija) {
91                          kirjoittaja.write(Integer.toString(saadutPisteet));
92                      } else {
93                          kirjoittaja.write(Integer.toString(kolmasSija));
94                      }
95                  }
96              }
97              
98              lukija.close();
99              kirjoittaja.close();
100         } catch (IOException | NumberFormatException e) {
101             System.out.print(e);
102         }
103     }
